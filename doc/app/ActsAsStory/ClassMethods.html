def acts_as_story(options = {})
  include ActsAsStory::InstanceMethods

  has_one :story, :as => :subject, :dependent => :destroy
  after_create :storify!

  cattr_accessor :story_fields
  self.story_fields = options.delete(:story_fields) { [] }
end
