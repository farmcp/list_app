def self.add_boston_sub_cities
  bos = City.find_by_name('Boston')
  bos_cities = [
    'cambridge',
    'lowell',
    'brockton',
    'lynn',
    'newton',
    'somerville',
    'lawrence',
    'quincy',
    'allston',
    'peabody',
    'nashua',
    'framingham',
    'waltham',
    'worcester',
  ]
  bos_cities.each do |sub_city|
    bos.sub_cities.create!(name: sub_city.downcase)
  end
end
