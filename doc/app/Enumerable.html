def histogram(&to_category_block)
  hist = Hash.new(0)
  each{ |i| hist[block_given? ? yield(i) : i] += 1 }
  hist
end

def variance
  return 0 if length == 0
  avg = mean
  sum = inject(0) { |acc,i| acc + (i - avg)**2 }
  (1 / length.to_f * sum)
end
